\documentclass{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[scale=2,transform shape] 
% First, center alignment -> wobbles 
\draw[anchor=center] (0,1) node{x} -- (0.5,1) 
    node{y} -- (1,1) node{t}; 
% Second, base alignment -> no wobble, but too high 
\draw[anchor=base] (0,.5) node{x} -- (0.5,.5) 
    node{y} -- (1,.5) node{t}; 
% Third, mid alignment 
\draw[anchor=mid] (0,0) node{x} -- (0.5,0) 
    node{y} -- (1,0) node{t}; 
\end{tikzpicture} 
\end{document}
